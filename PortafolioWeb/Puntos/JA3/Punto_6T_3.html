package mx.com.Etcetera;


public class Person {
	
	private String name;
	private String organization;
	private int tel;

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getOrganization() {
		return organization;
	}

	public void setOrganization(String organization) {
		this.organization = organization;
	}

	public int getTel() {
		return tel;
	}

	public void setTel(int tel) {
		this.tel = tel;
	}

	public Person(String name, String organization, int tel) {
		this.name = name;
		this.organization = organization;
		this.tel = tel;
	}

	public Person() {
	}
	
	
}
