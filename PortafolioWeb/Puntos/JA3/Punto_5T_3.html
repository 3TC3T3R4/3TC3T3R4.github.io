import java.util.Scanner;


public class Process {

private boolean option;
private int option2;

	public void getinfo(){
 
	option = true;
		
	}


	public  void showinfo() {
		
		while (option==true) {			
			
			
			System.out.println("1.Captura nombre\n2.Saludar persona\n3.Salir del sistema");
			option2=keyboard.nextInt();
			keyboard.nextLine();
		
		switch (option2) {
			case 1:
				
				System.out.println("Ingrese el nombre");
				text=keyboard.nextLine();
				
				System.out.println("Desea salir del programa?\n1.SI\n2.NO");
				option2=keyboard.nextInt();
				
				if(option2==1){
				
				option=false;
				
				}
				
				break;
			
			case 2:
			
				
				System.out.println("Ingrese el saludo");
				text=keyboard.nextLine();
				System.out.println("Desea salir del programa?\n1.SI\n2.NO");
				option2=keyboard.nextInt();
				
				if(option2==1){
				
				option=false;
				
				}
				
				
				break;
			
			case 3:
				
				option=false;
				
				break;
			
			
			default:
		}
		
			
			
	}
