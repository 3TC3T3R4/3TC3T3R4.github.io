Utilizando el ciclo que usted desee, crear un menú de ejecución infinita hasta que el usuario desee terminar dicho ciclo. 

		1.Capturar nombre
		2.Saludar persona
		3.Salir del sistema

	

		Proceso point5
	
		Definir option Como Entero;
		Definir op Como Logico;
		Definir text Como Caracter;
		op<-Verdadero;
		
		Mientras op == Verdadero Hacer
			
			Escribir "1.Capturar nombre";
			Escribir "2.Saludar persona";
			Escribir "3.Salir del sistema";
			Leer option;
			
			
			Segun option Hacer
				
				
				1:
					Escribir "Ingrese el nombre";
					leer text;
					Escribir "Desea salir del sistema?";
					Escribir "1. SI";
					Escribir "2. No";
					Leer option;
					
					Limpiar Pantalla;
					si option==1 Entonces
						op<-Falso;
					FinSi
					
					
				2:
					
					Escribir "Ingrese el saludo a enviar";
					Leer  text;
					Escribir "Desea salir del sistema?";
					Escribir "1. SI";
					Escribir "2. No";
					Leer option;
					Limpiar Pantalla;
					
					si option==1 Entonces
						op<-Falso;
					FinSi
					
				3:
					
				op <- Falso;
					
					
				De Otro Modo:
					Escribir"Ingrese un numero correcto";
				
				
			FinSegun
			
			
			
			
		FinMientras
		
		
		
		
		
		
	FinProceso
