public class Process {

	private final int matrix1[][] = new int[5][5];
	private final int matrix2[][] = new int[5][5];
	private int count;
	private int k;
	private int x;
	private int count2;
	private int i;
	private int j;

	public void getarray() {
		count = 1;
		count2 = 1;

		for (k = 0; k < 5; k++) {

			for (x = 0; x < 5; x++) {

				matrix2[k][x] = count2;
				count2 = count2 + 1;

			}

		}

		for (i = 0; i <= 3; i++) {

			if (i % 2 == 0) {

				for (j = 0; j <= 4; j++) {

					matrix1[i][j] = count;
					count = count + 1;

				}

			} else {

				for (j = 4; j >= 0; j--) {

					matrix1[i][j] = count;
					count = count + 1;

				}

			}

		}

	}

	public void showarray() {

		for (k = 0; k < 4; k++) {

			for (x = 0; x < 5; x++) {

				if (matrix2[k][x] >= 10) {

					System.out.print(" " + matrix2[k][x] + "");

				} else {

					System.out.print(" " + "0" + matrix2[k][x]);

				}

			}

			System.out.println("");

		}

		System.out.println("---------------------------------");
