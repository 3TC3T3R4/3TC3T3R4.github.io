package test;


public class Process {
	
	private final int array[] = new int [20];
	private int index;
	private int index2;
	
	
	public void numberArray(){
	
		
		
		for (index = 0; index < 20; index++) {
			
			 
			array[index]=(int) (Math.random() * 100);
		}
	
	
	}
	
	
	
	
	public void getarray(){
	
		index2=0;
		
		
		while (index2<20) {
			
			
			if (array[index2] % 2 ==0) {
			
				System.out.println("NUMERO PAR "+array[index2]);
				
				
			
			}else{
			
			
				System.out.println("NUMERO IMPAR "+array[index2]);
			
			
				
				
			}
			
			
			index2=index2+1;
		}
		
		
	
	
	}
	
}
