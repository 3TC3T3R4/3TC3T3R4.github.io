package test;

import java.util.Scanner;
import mx.com.Etcetera.Person;


public class ProcessImc {
	
	Scanner  keyboard = new Scanner(System.in);
	Person person1 = new Person();
	private float result;
	
	public void getinfo(){
	
		System.out.println("Ingrese su peso");
		person1.setWeight(keyboard.nextFloat());
		System.out.println("Ingrese su estatura");
		person1.setHeight(keyboard.nextFloat());
		
		result = person1.getHeight();
		
		result = (float)Math.pow(result,2);
		
		person1.setImc(person1.getWeight()/result);
	
	}
	
	
	public void showinfo(){
	
		if(person1.getImc()<18.5){
			
			System.out.println("Su indice de masa corporal es: "  + person1.getImc());
			System.out.println("Usted alctualmente esta en el rango: BAJO DE PESO");
			
		}else if(person1.getImc()>=18.5 && person1.getImc()<=24.9){
		
			
			System.out.println("Su indice de masa corporal es: "  + person1.getImc());
			System.out.println("Usted actualmente esta en el rango: NORMAL");
		
		
		
		}else if(person1.getImc()>=25.0 && person1.getImc()<=29.9){
		
			
			System.out.println("Su indice de masa corporal es: "  + person1.getImc());
			System.out.println("Usted actualmente esta en el rango: SOBREPESO");
		
		}else{
		
			System.out.println("Su indice de masa corporal es: "  + person1.getImc());
			System.out.println("Usted actualmente esta en el rango: OBESO");
		
		
		}
	
	
	}
	
}
