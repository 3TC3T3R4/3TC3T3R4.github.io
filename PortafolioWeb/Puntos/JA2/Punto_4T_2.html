import java.util.Scanner;

 */
public class Process {

	Scanner keyboard = new Scanner(System.in);
	Person person1 = new Person();
	private int option;
	private int option2;
	private String pqr;

	public void getinfo() {

		System.out.println("Bienvenido a nuestra tienda");
		System.out.println("Selecciona la accion que quieras realizar");
		System.out.println(" 1.Alquilar pelicula\n 2.Consultar peliculas disponibles\n 3.Hacer devolucion de peliculas");

		option = keyboard.nextInt();

	}

	public void showinfo() {

		switch (option) {
			case 1:
				System.out.println("Ingrese su nombre por favor");
				person1.setName(keyboard.nextLine());
				keyboard.nextLine();
				System.out.println(person1.getName() + "Las peliculas que puedes alquilar son estas");
				System.out.println("1.The walking dead\n 2.Harry Potter\n 3. BlackMirror");
				option2 = keyboard.nextInt();

				if (option2 == 1 || option2 == 2 || option2 == 3) {

					System.out.println("Excelente opcion disfruta tu pelicula");

				} else {

					System.out.println("Selecciona una opcion correcta");

				}

				break;

			case 2:

				System.out.println("Estas son las peliculas  disponibles por el momento\n--The walking dead\n  --Harry Potter\n --BlackMirror");
				break;

			case 3:
				System.out.println("Gracias por tu devolucion");
				System.out.println("Tuviste algun inconveniente?\n 1.Si quiero presentar novedad\n 2.No se presento ninguna novedad");
				option2 = keyboard.nextInt();

				if (option2 == 1) {
					System.out.println("Ingresa tu nombre para que la novedad sea tuya");
					person1.setName(keyboard.nextLine());
					keyboard.nextLine();
					System.out.println("Cuentanos que paso?");
					pqr = keyboard.nextLine();
					keyboard.nextLine();
					System.out.println("Gracias por la observacion trabajaremos en ello");

				} else {

					System.out.println("Esta bien gracias");

				}

		}

	}

}
