package mx.com.Etecetera;

import java.util.Scanner;

/**
 *
 * @author 4dm1n1str4t0r
 */
public class Person {

	Scanner keyboard = new Scanner(System.in);
	private String name;
	private String lastname;
	private int age;

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getLastname() {
		return lastname;
	}

	public void setLastname(String lastname) {
		this.lastname = lastname;
	}

	public int getAge() {
		return age;
	}

	public void setAge(int age) {
		this.age = age;
	}

	public void showinfo() {

		System.out.println("Ingrese su nombre");
		name= keyboard.nextLine();
		System.out.println("Ingrese su apellido");
		lastname=keyboard.nextLine();
		System.out.println("Ingrese su edad");
		age=keyboard.nextInt();
		if (age >= 18) {

			System.out.println(name + " " + lastname + " Usted es mayor de edad por lo tanto puede entrar a la fiesta");

		} else {

			System.out.println(name + " " + lastname + " Usted es menor de edad, por lo tanto no puede entrar a la fiesta,"
				+ " devuelvase a su casa");

		}

	}

}
